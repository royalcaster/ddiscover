  \begin{slidecanvas}

    \node[anchor=north west,text=AppMuted,font=\scriptsize,text width=105mm] at (1.0,4.3) {
      The standalone APK exposed behavior that did not appear in local development builds.
    };

    \node[anchor=north west,text=AppText,font=\scriptsize\bfseries,text width=28mm] at (1.0,3.45) {Observed};
    \node[anchor=north west,text=AppMuted,font=\scriptsize,text width=80mm] at (4.0,3.45) {
      Metro and the dev client loaded data correctly, but the standalone APK failed on Convex data access.
    };

    \node[anchor=north west,text=AppText,font=\scriptsize\bfseries,text width=28mm] at (1.0,2.55) {Root causes};
    \node[anchor=north west,text=AppMuted,font=\scriptsize,text width=80mm] at (4.0,2.55) {
      Unstable Convex function references in effects, WebSocket behavior in the APK context, and a trailing slash in the CI Convex URL.
    };

    \node[anchor=north west,text=AppText,font=\scriptsize\bfseries,text width=28mm] at (1.0,1.45) {Fix};
    \node[anchor=north west,text=AppMuted,font=\scriptsize,text width=80mm] at (4.0,1.45) {
      Public app data loads through stable Convex HTTP queries; runtime config strips trailing slashes before building API paths.
    };
  \end{slidecanvas}
